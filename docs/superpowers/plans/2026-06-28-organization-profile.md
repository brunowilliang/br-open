# Organization Profile Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use
> superpowers:subagent-driven-development (recommended) or
> superpowers:executing-plans to implement this plan task-by-task. Steps use
> checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the organization from a blind fallback record into a real,
editable organizer profile with a dedicated onboarding flow and an edit screen.

**Architecture:** The `organization` table (Better Auth-managed) already has
`name`, `slug`, `logo`, and a `metadata` json column. This slice types that
`metadata` with zod, adds a new `organization` domain (mirroring `player`) with
`get`/`upsert`/`generateUploadUrl` functions, refactors `activateOrganization`
to accept a rich onboarding payload, and adds two new routes (`onboarding.tsx`,
`profile.tsx`) plus a dynamic "Perfil" entry in Settings. No new table; no
migration; no payment.

**Tech Stack:** Convex + kitcn CRPC, zod, React Native + Expo Router, HeroUI
Native, react-hook-form, React Query, ViaCEP (client-side CEP lookup).

**Spec:** `docs/superpowers/specs/2026-06-28-organization-profile-design.md`

```ts
import { z } from "zod";
import { requiredString } from "../../utils/contract.zod";

export const ORGANIZER_TYPES = [
  "academia",
  "clube",
  "liga",
  "condominio",
  "escola",
  "federacao",
  "confederacao",
  "centro_de_treinamento",
  "outro",
] as const;

export type OrganizerType = (typeof ORGANIZER_TYPES)[number];

const PHYSICAL_ORGANIZER_TYPES = [
  "academia",
  "clube",
  "centro_de_treinamento",
  "escola",
  "condominio",
] as const;

const PHYSICAL_ORGANIZER_TYPE_SET = new Set<string>(PHYSICAL_ORGANIZER_TYPES);

export function isPhysicalOrganizationType(type?: null | string) {
  return type !== undefined && type !== null &&
    PHYSICAL_ORGANIZER_TYPE_SET.has(type);
}

export const SPORTS = [
  "tenis",
  "beach_tennis",
  "futevolei",
  "volei_de_praia",
  "padel",
  "squash",
  "futebol_society",
  "pickleball",
  "tenis_de_mesa",
  "raquetinha",
  "badminton",
  "volei_de_quadra",
  "outro",
] as const;

export type OrganizationSport = (typeof SPORTS)[number];

const organizerTypeEnum = z.enum(ORGANIZER_TYPES, {
  error: (issue) => issue.input === undefined ? "Selecione o tipo." : undefined,
});

const sportsEnum = z.enum(SPORTS);

export const addressSchema = z.object({
  cep: requiredString("Informe o CEP."),
  street: z.string(),
  district: z.string().optional(),
  city: z.string(),
  state: z.string(),
  number: requiredString("Informe o número."),
  complement: z.string().optional(),
});

export type OrganizationAddress = z.infer<typeof addressSchema>;

const acceptedTermsSchema = z.object({
  version: z.string().min(1),
  acceptedAt: z.string().min(1),
  userId: z.string().min(1),
});

export type AcceptedTerms = z.infer<typeof acceptedTermsSchema>;

const logoStorageIdSchema = z.string().min(1, "Imagem inválida.").nullable();

export const organizationMetadataSchema = z
  .object({
    organizerType: organizerTypeEnum.nullable().optional(),
    address: addressSchema.nullable().optional(),
    sports: z.array(sportsEnum).nullable().optional(),
    description: z.string().trim().nullable().optional(),
    website: z.string().trim().nullable().optional(),
    contactEmail: z.string().trim().nullable().optional(),
    acceptedTerms: acceptedTermsSchema.nullable().optional(),
  })
  // Legacy organizations may have stored `{}` or `null`; normalize to `{}`.
  .nullish()
  .transform((value) => value ?? {})
  .pipe(
    z.object({
      organizerType: organizerTypeEnum.nullable().optional(),
      address: addressSchema.nullable().optional(),
      sports: z.array(sportsEnum).nullable().optional(),
      description: z.string().trim().nullable().optional(),
      website: z.string().trim().nullable().optional(),
      contactEmail: z.string().trim().nullable().optional(),
      acceptedTerms: acceptedTermsSchema.nullable().optional(),
    }),
  );

export type OrganizationMetadata = z.infer<typeof organizationMetadataSchema>;

const baseOrganizationFields = {
  name: requiredString("Informe o nome da organização."),
  logoStorageId: logoStorageIdSchema,
  description: z.string().trim().optional(),
  website: z.string().trim().optional(),
  contactEmail: z.string().trim().optional(),
  sports: z.array(sportsEnum).optional(),
};

/**
 * Enforces the coupling between `organizerType` and `address`:
 * physical types require an address; non-physical types must not include one.
 */
function refineOrganizerAddress(input: {
  address?: unknown;
  organizerType?: null | string;
}) {
  return z
    .any()
    .superRefine((_, ctx) => {
      const isPhysical = isPhysicalOrganizationType(input.organizerType);
      const hasAddress = input.address !== undefined &&
        input.address !== null &&
        !(typeof input.address === "object" && input.address !== null &&
          Object.keys(input.address).length === 0);

      if (isPhysical && !hasAddress) {
        ctx.addIssue({
          code: z.ZodIssueCode.custom,
          message: "Informe o endereço da sede.",
          path: ["address"],
        });
      }

      if (!isPhysical && hasAddress) {
        ctx.addIssue({
          code: z.ZodIssueCode.custom,
          message: "Este tipo de organização não possui endereço físico.",
          path: ["address"],
        });
      }
    });
}

export const activateOrganizationSchema = z
  .object({
    ...baseOrganizationFields,
    organizerType: organizerTypeEnum,
    address: addressSchema.nullable().optional(),
    acceptedTerms: acceptedTermsSchema,
  })
  .superRefine((value, ctx) => {
    const result = refineOrganizerAddress(value).safeParse({});

    if (!result.success) {
      for (const issue of result.error.issues) {
        ctx.addIssue(issue);
      }
    }
  });

export type ActivateOrganizationInput = z.infer<
  typeof activateOrganizationSchema
>;

export const upsertOrganizationSchema = z
  .object({
    ...baseOrganizationFields,
    organizerType: organizerTypeEnum,
    address: addressSchema.nullable().optional(),
  })
  .superRefine((value, ctx) => {
    const result = refineOrganizerAddress(value).safeParse({});

    if (!result.success) {
      for (const issue of result.error.issues) {
        ctx.addIssue(issue);
      }
    }
  })
  // Editing never re-accepts terms: the field is deliberately absent so any
  // caller that passes it is rejected.
  .refine(
    () => true,
    "upsert never accepts acceptedTerms; if you see this the schema leaked.",
  );

export type UpsertOrganizationInput = z.infer<typeof upsertOrganizationSchema>;

export const organizationOutputSchema = z.object({
  id: z.string(),
  name: z.string(),
  slug: z.string(),
  logoStorageId: logoStorageIdSchema,
  logoUrl: z.string().nullable().optional(),
  organizerType: organizerTypeEnum.nullable().optional(),
  address: addressSchema.nullable().optional(),
  sports: z.array(sportsEnum).nullable().optional(),
  description: z.string().nullable().optional(),
  website: z.string().nullable().optional(),
  contactEmail: z.string().nullable().optional(),
  acceptedTerms: acceptedTermsSchema.nullable().optional(),
});

export type OrganizationOutput = z.infer<typeof organizationOutputSchema>;

export function collectReplacedLogoStorageIds(input: {
  next: { logoStorageId?: null | string };
  previous?: { logoStorageId?: null | string } | null;
}) {
  const previousLogo = input.previous?.logoStorageId ?? null;
  const nextLogo = input.next.logoStorageId ?? null;

  if (!previousLogo || previousLogo === nextLogo) {
    return [];
  }

  return [previousLogo];
}

export const CURRENT_TERMS_VERSION = "1.0.0";
```

```ts
import {
  organizationMetadataSchema,
  type OrganizationOutput,
  organizationOutputSchema,
} from "./contract";

const ORGANIZATION_NAME_FALLBACK = "Minha organização";

export function buildOrganizationDisplayName(input: {
  name?: null | string;
}) {
  const trimmed = input.name?.trim();

  if (trimmed) {
    return trimmed;
  }

  return ORGANIZATION_NAME_FALLBACK;
}

type OrganizationRecord = {
  createdAt: number;
  id: string;
  logo?: null | string;
  metadata?: null | Record<string, unknown>;
  name: string;
  slug: string;
  updatedAt?: null | number;
};

export async function serializeOrganizationOutput(input: {
  logoUrlResolver: (storageId: string) => Promise<null | string>;
  record: OrganizationRecord;
}): Promise<OrganizationOutput> {
  const logoStorageId = input.record.logo ?? null;
  const logoUrl = logoStorageId
    ? await input.logoUrlResolver(logoStorageId).catch(() => null)
    : null;
  const metadata = organizationMetadataSchema.parse(input.record.metadata);

  return organizationOutputSchema.parse({
    id: input.record.id,
    logoStorageId,
    logoUrl,
    name: buildOrganizationDisplayName({ name: input.record.name }),
    slug: input.record.slug,
    ...metadata,
  });
}
```

```ts
import { CRPCError } from "kitcn/server";
import type { InferSelectModel } from "kitcn/orm";
import { z } from "zod";
import type { Id } from "../../functions/_generated/dataModel";
import type { MutationCtx, QueryCtx } from "../../functions/generated/server";
import { organization } from "../../domains/auth/tables";
import {
  collectReplacedLogoStorageIds,
  organizationOutputSchema,
  upsertOrganizationSchema,
} from "../../domains/organization/contract";
import { serializeOrganizationOutput } from "../../domains/organization/identity";
import { type AuthenticatedCtx, authMutation, authQuery } from "../../lib/crpc";
import {
  getViewerContext,
  requireActiveManager,
  requireActiveOrganization,
} from "../viewer/context";

type OrganizationRecord = InferSelectModel<typeof organization>;
type ViewerCtx = AuthenticatedCtx<QueryCtx | MutationCtx>;
type ViewerMutationCtx = AuthenticatedCtx<MutationCtx>;

async function resolveLogoUrl(
  ctx: QueryCtx | MutationCtx,
  storageId?: null | string,
) {
  if (!storageId) {
    return null;
  }

  try {
    return await ctx.storage.getUrl(storageId as Id<"_storage">);
  } catch {
    return null;
  }
}

async function serializeOrganization(
  ctx: QueryCtx | MutationCtx,
  record: OrganizationRecord,
) {
  return serializeOrganizationOutput({
    logoUrlResolver: (storageId) => resolveLogoUrl(ctx, storageId),
    record,
  });
}

async function deleteLogoStorageIds(ctx: MutationCtx, storageIds: string[]) {
  for (const storageId of storageIds) {
    await ctx.storage.delete(storageId as Id<"_storage">);
  }
}

function buildMetadataFromInput(input: {
  address?: z.infer<typeof upsertOrganizationSchema.shape.address> | null;
  contactEmail?: string;
  description?: string;
  sports?: z.infer<typeof upsertOrganizationSchema.shape.sports>;
  website?: string;
}) {
  return {
    address: input.address ?? undefined,
    contactEmail: input.contactEmail || undefined,
    description: input.description || undefined,
    sports: input.sports,
    website: input.website || undefined,
  };
}

export const get = authQuery
  .output(organizationOutputSchema.nullable())
  .query(async ({ ctx }) => {
    const organizationId = await requireActiveOrganization(ctx);
    const record = await ctx.orm.query.organization.findFirst({
      where: (org, { eq }) => eq(org.id, organizationId),
    });

    if (!record) {
      return null;
    }

    return serializeOrganization(ctx, record);
  });

export const generateUploadUrl = authMutation
  .output(z.string())
  .mutation(async ({ ctx }) => {
    await requireActiveManager(ctx);
    return ctx.storage.generateUploadUrl();
  });

export const upsert = authMutation
  .input(upsertOrganizationSchema)
  .output(organizationOutputSchema)
  .mutation(async ({ ctx, input }) => {
    const organizationId = await requireActiveManager(ctx);
    const current = await ctx.orm.query.organization.findFirst({
      where: (org, { eq }) => eq(org.id, organizationId),
    });

    if (!current) {
      throw new CRPCError({
        code: "NOT_FOUND",
        message: "Organização não encontrada.",
      });
    }

    const replacedStorageIds = collectReplacedLogoStorageIds({
      next: { logoStorageId: input.logoStorageId },
      previous: { logoStorageId: current.logo },
    });

    const metadata = {
      ...(current.metadata ?? {}),
      ...buildMetadataFromInput(input),
      organizerType: input.organizerType,
    };

    await ctx.orm
      .update(organization)
      .set({
        logo: input.logoStorageId,
        metadata,
        name: input.name,
        updatedAt: new Date(),
      })
      .where((org, { eq }) => eq(org.id, organizationId));

    await deleteLogoStorageIds(ctx, replacedStorageIds);

    const next = await ctx.orm.query.organization.findFirst({
      where: (org, { eq }) => eq(org.id, organizationId),
    });

    if (!next) {
      throw new Error("Organization was not found after upsert.");
    }

    return serializeOrganization(ctx, next);
  });

// Re-export so the activation mutation (Task 4) can reuse serialization.
export { getViewerContext };
```

- [ ] **Step 1: Create the helper**

```ts
export type ViaCepResult = {
  cep: string;
  street: string;
  district: string;
  city: string;
  state: string;
};

export class ViaCepNotFoundError extends Error {
  constructor() {
    super("CEP não encontrado.");
  }
}

export class ViaCepFetchError extends Error {
  constructor() {
    super("Não foi possível buscar o CEP.");
  }
}

const VIA_CEP_BASE = "https://viacep.com.br/ws";

function sanitizeCep(cep: string) {
  return cep.replace(/\D/g, "");
}

export async function fetchAddressByCep(
  rawCep: string,
): Promise<ViaCepResult> {
  const cep = sanitizeCep(rawCep);

  if (cep.length !== 8) {
    throw new ViaCepFetchError();
  }

  let response: Response;

  try {
    response = await fetch(`${VIA_CEP_BASE}/${cep}/json/`);
  } catch {
    throw new ViaCepFetchError();
  }

  if (!response.ok) {
    throw new ViaCepFetchError();
  }

  let body: unknown;

  try {
    body = await response.json();
  } catch {
    throw new ViaCepFetchError();
  }

  if (
    typeof body !== "object" ||
    body === null ||
    "erro" in body ||
    !("localidade" in body) ||
    !("uf" in body)
  ) {
    throw new ViaCepNotFoundError();
  }

  const data = body as Record<string, string>;

  return {
    cep: data.cep ?? rawCep,
    city: data.localidade ?? "",
    district: data.bairro ?? "",
    state: data.uf ?? "",
    street: data.logradouro ?? "",
  };
}
```

## Notes for the Implementer

- **`convex/shared/api.ts` is generated.** Never hand-edit it. Run
  `bun run codegen` after creating or changing any function file in
  `convex/functions/`.
- **No DB-mock test helper exists in this repo.** Tests are contract-level (zod
  parse) or pure-function tests. The mutation functions in
  `convex/functions/organization/profile.ts` are not unit-tested, matching the
  existing `player/profile.ts` precedent.
- **`requireActiveOrganization` / `requireActiveManager`** already exist in
  `convex/functions/viewer/context.ts` — reuse them, do not recreate.
- **ViaCEP is a UX convenience only.** The server stores whatever the client
  sends; do not add server-side CEP validation in this slice.
- **`acceptedTerms` is read-only after activation.** The
  `upsertOrganizationSchema` deliberately omits the field so editing never
  re-prompts for terms.
- **The `organization` table is Better Auth-managed.** We only read/write
  `name`, `logo`, and `metadata`. Do not add columns to it in this slice;
  promote-on-demand only if a future search feature needs it.
